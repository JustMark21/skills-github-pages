        // --- RENDER FUNCTION ---

        function renderGrid() {

            grid.innerHTML = '';

            

            const filteredItems = catalogItems.filter(item => 

                item.name.toLowerCase().includes(state.searchQuery) || 

                item.id.toString().includes(state.searchQuery)

            );


            if (filteredItems.length === 0) {

                noResults.classList.remove('hidden');

            } else {

                noResults.classList.add('hidden');

            }


            filteredItems.forEach(item => {

                const isSelected = state.selectedIds.has(item.id);

                

                const card = document.createElement('div');

                card.className = `item-card bg-white rounded-lg shadow border border-gray-100 overflow-hidden flex flex-col ${isSelected ? 'selected' : ''}`;

                card.onclick = (e) => {

                    // Prevent triggering selection when clicking the "View Image" link

                    if(e.target.tagName !== 'A' && e.target.parentElement.tagName !== 'A') {

                        toggleSelection(item.id);

                    }

                };


                // Image Element with Error Fallback

        // --- ACTIONS ---

        function toggleSelection(id) {

            if (state.selectedIds.has(id)) {

                state.selectedIds.delete(id);

            } else {

                state.selectedIds.add(id);

            }

            updateUI();

            renderGrid(); // Re-render to update styling

        }


        function updateUI() {

            const count = state.selectedIds.size;

            selectedCountLabel.innerText = count;

            

            // Show/Hide selection bar

            if (count > 0) {

                selectionBar.classList.remove('translate-y-full');

            } else {

                selectionBar.classList.add('translate-y-full');

            }


            // Update Form Textarea

            const selectedItemsArray = catalogItems.filter(item => state.selectedIds.has(item.id));

            if (selectedItemsArray.length > 0) {

                const listText = selectedItemsArray.map(item => `[#${item.id}] ${item.name}`).join('\n');

                selectedItemsInput.value = listText;

            } else {

                selectedItemsInput.value = '';

            }

        }


        function scrollToForm() {

            document.getElementById('requestFormSection').scrollIntoView({ behavior: 'smooth' });

        }


        function handleFormSubmit(e) {

            e.preventDefault();

            const formData = new FormData(e.target);

            const name = formData.get('name');

            const items = formData.get('selected_items');

            const notes = formData.get('notes');


            // Construct email body

            const subject = `Catalog Order from ${name}`;

            const body = `Name: ${name}\n\nSelected Items:\n${items}\n\nNotes:\n${notes}`;

            

            // Open Mail Client

            window.location.href = `mailto:?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(body)}`;

            

            // Optional: Alert user

            alert("Your mail client has been opened with your selection!");

        }
